# Define your Shiny server logic here
server <- function(input, output, session) {

  output$selectedStudies <- renderDataTable({
    
    # Select studies / conditions
    st <- subset(st, st$Experimental == input$condition)
    
    # Check if no checkboxes are selected
    if (length(input$labOnline) == 0) {
      # If none are selected, return an empty data frame
      return(data.frame())
    } else {
      # Filter the data based on the selected checkboxes
      st <- subset(st, st$Labstudy %in% input$labOnline)
    }
    
    st
    
  })

}


# Create and launch the Shiny app
shinyApp(ui, server)
